# Vote Window Overrides Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Allow superadmin to extend the BNF/CV voting window for an entire round or a specific fixture, and club_admin to extend it for any fixture involving their team.

**Architecture:** A new `vote_window_overrides` DB table (keyed on competition/grade/round + optional fixtureId) drives all window checks. A shared `lib/voteWindow.ts` helper resolves the effective window for any fixture — match-level override beats round-level beats the global `VOTE_WINDOW` constant. Four existing API routes (CV and BNF fixtures + submission) swap their inline date checks for this helper. A new `/admin/vote-windows` page (visible to all admins) provides CRUD.

**Tech Stack:** Next.js App Router, Drizzle ORM (SQLite/libsql), NextAuth v5, TypeScript, CSS Modules

### Task 2: Create `lib/voteWindow.ts` — shared window resolution helper

**Files:**
- Create: `lib/voteWindow.ts`

This is the single place that resolves "is this fixture currently open for voting?" — used by all four API routes.

- [ ] **Step 1: Create the file**

```ts
import { db } from "@/lib/db";
import { voteWindowOverrides } from "@/db/schema";
import { and, eq, isNull, desc } from "drizzle-orm";
import { VOTE_WINDOW } from "@/lib/constants";

function getTasDate(offsetDays = 0): string {
  const d = new Date();
  d.setDate(d.getDate() + offsetDays);
  return new Intl.DateTimeFormat("en-CA", {
    timeZone: "Australia/Hobart",
    year: "numeric", month: "2-digit", day: "2-digit",
  }).format(d);
}

export type VoteWindowResult = {
  inWindow:      boolean;
  extendedUntil: string | null; // set when an active override applies
};

/**
 * Resolves the effective voting window for a fixture.
 *
 * Resolution order (most specific wins):
 *   1. Match-level override (by fixtureId)
 *   2. Round-level override (by competition + grade + round, fixtureId IS NULL)
 *   3. Global VOTE_WINDOW constant in lib/constants.ts
 *
 * @param matchDate   - YYYY-MM-DD date the match was played
 * @param competition - "SFL" | "STJFL"
 * @param grade       - grade name e.g. "SFL Community League Senior Men"
 * @param round       - e.g. "Round 5"
 * @param fixtureId   - PlayHQ fixture ID; null if fixture couldn't be found
 */
export async function resolveVoteWindow(
  matchDate:   string,
  competition: string,
  grade:       string,
  round:       string,
  fixtureId:   string | null,
): Promise<VoteWindowResult> {
  if (!VOTE_WINDOW.enforce) return { inWindow: true, extendedUntil: null };

  const today = getTasDate(0);

  // Match hasn't been played yet — never in window
  if (matchDate > today) return { inWindow: false, extendedUntil: null };

  // 1. Match-level override — ORDER BY id DESC so the most recently upserted row wins
  //    if duplicates somehow exist (SQLite serialises writes so this is unlikely).
  if (fixtureId) {
    const [matchOverride] = await db
      .select({ extendedUntil: voteWindowOverrides.extendedUntil })
      .from(voteWindowOverrides)
      .where(eq(voteWindowOverrides.fixtureId, fixtureId))
      .orderBy(desc(voteWindowOverrides.id))
      .limit(1);

    if (matchOverride) {
      return {
        inWindow:      today <= matchOverride.extendedUntil,
        extendedUntil: matchOverride.extendedUntil,
      };
    }
  }

  // 2. Round-level override
  const [roundOverride] = await db
    .select({ extendedUntil: voteWindowOverrides.extendedUntil })
    .from(voteWindowOverrides)
    .where(
      and(
        eq(voteWindowOverrides.competition, competition),
        eq(voteWindowOverrides.grade,       grade),
        eq(voteWindowOverrides.round,       round),
        isNull(voteWindowOverrides.fixtureId),
      )
    )
    .orderBy(desc(voteWindowOverrides.id))
    .limit(1);

  if (roundOverride) {
    return {
      inWindow:      today <= roundOverride.extendedUntil,
      extendedUntil: roundOverride.extendedUntil,
    };
  }

  // 3. Default window
  const earliest = getTasDate(-VOTE_WINDOW.daysAfterMatch);
  return {
    inWindow:      matchDate >= earliest,
    extendedUntil: null,
  };
}
```

Auth rules for the API:
- Both `superadmin` and `club_admin` may access this endpoint.
- **GET**: superadmin sees all overrides; club_admin sees only overrides for their `scopedGrades`.
- **POST**: superadmin can create round-level or match-level for any grade; club_admin can only create match-level for fixtures involving their club's teams (looked up via `clubId` → `teams`).
- **DELETE**: superadmin can delete any; club_admin can only delete overrides they created (`createdBy === Number(session.user.id)`).

- [ ] **Step 4: Create the file**

```ts
import { NextRequest, NextResponse } from "next/server";
import { and, eq, isNull } from "drizzle-orm";

import { auth } from "@/auth";
import { db } from "@/lib/db";
import { voteWindowOverrides, fixtures, teams } from "@/db/schema";
import { ROUND_OPTIONS, COMPETITIONS } from "@/lib/constants";
import { logger } from "@/lib/logger";

const DATE_RE = /^\d{4}-\d{2}-\d{2}$/;
const ROUND_SET = new Set(ROUND_OPTIONS);
const COMP_SET  = new Set(COMPETITIONS);

function err(msg: string, status = 400) {
  return NextResponse.json({ error: msg }, { status });
}

// ─── GET /api/admin/vote-window-overrides ─────────────────────────────────────
export async function GET() {
  const session = await auth();
  if (!session) return err("Unauthorized", 401);
  const { role, scopedGrades } = session.user;
  if (role !== "superadmin" && role !== "club_admin") return err("Forbidden", 403);

  try {
    const rows = await db
      .select({
        id:            voteWindowOverrides.id,
        competition:   voteWindowOverrides.competition,
        grade:         voteWindowOverrides.grade,
        round:         voteWindowOverrides.round,
        fixtureId:     voteWindowOverrides.fixtureId,
        homeTeamName:  fixtures.homeTeamName,
        awayTeamName:  fixtures.awayTeamName,
        extendedUntil: voteWindowOverrides.extendedUntil,
        createdBy:     voteWindowOverrides.createdBy,
        createdAt:     voteWindowOverrides.createdAt,
      })
      .from(voteWindowOverrides)
      .leftJoin(fixtures, eq(voteWindowOverrides.fixtureId, fixtures.id))
      .orderBy(voteWindowOverrides.createdAt);

    const filtered = role === "club_admin" && scopedGrades
      ? rows.filter((r) => scopedGrades.includes(r.grade))
      : rows;

    const result = filtered.map((r) => ({
      ...r,
      fixtureLabel: r.homeTeamName && r.awayTeamName
        ? `${r.homeTeamName} vs ${r.awayTeamName}`
        : null,
    }));

    return NextResponse.json(result);
  } catch (e) {
    logger.error("[vote-window-overrides] GET failed", { category: "api", error: String(e) });
    return err("Internal server error.", 500);
  }
}

// ─── POST /api/admin/vote-window-overrides ────────────────────────────────────
export async function POST(req: NextRequest) {
  const session = await auth();
  if (!session) return err("Unauthorized", 401);
  const { role, scopedGrades } = session.user;
  if (role !== "superadmin" && role !== "club_admin") return err("Forbidden", 403);

  let body: unknown;
  try { body = await req.json(); } catch { return err("Invalid JSON body."); }
  if (typeof body !== "object" || body === null) return err("Body must be a JSON object.");
  const b = body as Record<string, unknown>;

  const competition   = typeof b.competition   === "string" ? b.competition.trim()   : null;
  const grade         = typeof b.grade         === "string" ? b.grade.trim()         : null;
  const round         = typeof b.round         === "string" ? b.round.trim()         : null;
  const fixtureId     = typeof b.fixtureId     === "string" ? b.fixtureId.trim()     : null;
  const extendedUntil = typeof b.extendedUntil === "string" ? b.extendedUntil.trim() : null;

  if (!competition)   return err("competition is required.");
  if (!grade)         return err("grade is required.");
  if (!round)         return err("round is required.");
  if (!extendedUntil) return err("extendedUntil is required.");

  if (!COMP_SET.has(competition))  return err(`Unknown competition: "${competition}".`);
  if (!ROUND_SET.has(round))       return err(`Unknown round: "${round}".`);
  if (!DATE_RE.test(extendedUntil) || isNaN(new Date(extendedUntil).getTime())) {
    return err("extendedUntil must be a valid YYYY-MM-DD date.");
  }

  // club_admin: round-level overrides not allowed
  if (role === "club_admin" && fixtureId === null) {
    return err("club_admin may only create match-level overrides.", 403);
  }

  // club_admin: fixture must involve one of their club's teams
  if (role === "club_admin") {
    const { clubId } = session.user;
    if (!clubId) return err("Your account has no associated club.", 403);

    // Look up team names for this club
    const clubTeams = await db
      .select({ teamName: teams.teamName })
      .from(teams)
      .where(eq(teams.clubId, clubId));

    const clubTeamNames = new Set(clubTeams.map((t) => t.teamName));
    if (clubTeamNames.size === 0) return err("No teams found for your club.", 403);

    if (!fixtureId) return err("club_admin must select a specific fixture.", 403);

    const [fix] = await db
      .select({ homeTeamName: fixtures.homeTeamName, awayTeamName: fixtures.awayTeamName })
      .from(fixtures)
      .where(eq(fixtures.id, fixtureId))
      .limit(1);

    if (!fix) return err(`Fixture "${fixtureId}" not found.`, 404);

    if (!clubTeamNames.has(fix.homeTeamName) && !clubTeamNames.has(fix.awayTeamName)) {
      return err("You may only create overrides for fixtures involving your club's teams.", 403);
    }
  }

  // For superadmin: verify the fixture exists and belongs to the stated grade/round
  if (fixtureId && role === "superadmin") {
    const [fix] = await db
      .select({ gradeName: fixtures.gradeName, roundName: fixtures.roundName })
      .from(fixtures)
      .where(eq(fixtures.id, fixtureId))
      .limit(1);

    if (!fix) return err(`Fixture "${fixtureId}" not found.`, 404);
    if (fix.gradeName !== grade) return err("Fixture does not belong to the selected grade.");
    if (fix.roundName !== round) return err("Fixture does not belong to the selected round.");
  }

  // Upsert: if an override already exists for this key, replace it
  const existing = await db
    .select({ id: voteWindowOverrides.id })
    .from(voteWindowOverrides)
    .where(
      fixtureId
        ? eq(voteWindowOverrides.fixtureId, fixtureId)
        : and(
            eq(voteWindowOverrides.competition, competition),
            eq(voteWindowOverrides.grade,       grade),
            eq(voteWindowOverrides.round,       round),
            isNull(voteWindowOverrides.fixtureId),
          )
    )
    .limit(1);

  try {
    if (existing.length > 0) {
      await db
        .update(voteWindowOverrides)
        .set({ extendedUntil, createdBy: Number(session.user.id), createdAt: new Date().toISOString() })
        .where(eq(voteWindowOverrides.id, existing[0].id));

      logger.info("[vote-window-overrides] updated", { category: "business", grade, round, fixtureId, extendedUntil });
      return NextResponse.json({ success: true, id: existing[0].id });
    }

    const [inserted] = await db
      .insert(voteWindowOverrides)
      .values({
        competition, grade, round,
        fixtureId:     fixtureId ?? null,
        extendedUntil,
        createdBy:     Number(session.user.id),
      })
      .returning();

    logger.info("[vote-window-overrides] created", { category: "business", grade, round, fixtureId, extendedUntil });
    return NextResponse.json({ success: true, id: inserted.id }, { status: 201 });
  } catch (e) {
    logger.error("[vote-window-overrides] POST failed", { category: "api", error: String(e) });
    return err("Internal server error.", 500);
  }
}

// ─── DELETE /api/admin/vote-window-overrides?id=123 ───────────────────────────
export async function DELETE(req: NextRequest) {
  const session = await auth();
  if (!session) return err("Unauthorized", 401);
  const { role } = session.user;
  if (role !== "superadmin" && role !== "club_admin") return err("Forbidden", 403);

  const idParam = req.nextUrl.searchParams.get("id");
  const id = idParam ? parseInt(idParam, 10) : NaN;
  if (isNaN(id)) return err("id query param is required and must be a number.");

  try {
    const [row] = await db
      .select({ id: voteWindowOverrides.id, createdBy: voteWindowOverrides.createdBy })
      .from(voteWindowOverrides)
      .where(eq(voteWindowOverrides.id, id))
      .limit(1);

    if (!row) return err("Override not found.", 404);

    // club_admin can only delete their own overrides
    if (role === "club_admin" && row.createdBy !== Number(session.user.id)) {
      return err("You may only delete overrides you created.", 403);
    }

    await db.delete(voteWindowOverrides).where(eq(voteWindowOverrides.id, id));

    logger.info("[vote-window-overrides] deleted", { category: "business", id });
    return NextResponse.json({ success: true });
  } catch (e) {
    logger.error("[vote-window-overrides] DELETE failed", { category: "api", error: String(e) });
    return err("Internal server error.", 500);
  }
}
```

```tsx
"use client";

import { useState, useEffect, useCallback } from "react";
import { useSession } from "next-auth/react";
import styles from "./vote-windows.module.css";
import { COMPETITIONS, GRADE_MAP, ROUND_OPTIONS } from "@/lib/constants";

type Override = {
  id:            number;
  competition:   string;
  grade:         string;
  round:         string;
  fixtureId:     string | null;
  fixtureLabel:  string | null;
  extendedUntil: string;
  createdBy:     number;
  createdAt:     string;
};

type FixtureOption = {
  id:           string;
  homeTeamName: string;
  awayTeamName: string;
  matchDate:    string;
};

function allGradesFor(competition: string): string[] {
  return Object.entries(GRADE_MAP)
    .filter(([key]) => key.startsWith(competition))
    .flatMap(([, grades]) => grades);
}

export default function VoteWindowsPage() {
  const { data: session } = useSession();
  const isSuperadmin = session?.user?.role === "superadmin";
  const scopedGrades = session?.user?.scopedGrades ?? null;

  const [overrides, setOverrides] = useState<Override[]>([]);

  const [extendedUntil, setExtendedUntil] = useState("");
  const [fixtures,      setFixtures]      = useState<FixtureOption[]>([]);

  const [formError,     setFormError]     = useState<string | null>(null);

  const grades = isSuperadmin
    ? allGradesFor(competition)
    : (scopedGrades ?? []);

  const loadOverrides = useCallback(async () => {
    setLoading(true);
    try {
      const data = await fetch("/api/admin/vote-window-overrides").then((r) => r.json());
      setOverrides(Array.isArray(data) ? data : []);
    } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => { loadOverrides(); }, [loadOverrides]);

  // Load fixtures when grade + round are selected
  useEffect(() => {
    if (!grade || !round) { setFixtures([]); return; }
    fetch(`/api/fixtures?grade=${encodeURIComponent(grade)}&round=${encodeURIComponent(round)}`)
      .then((r) => r.json())
      .then((data) => setFixtures(Array.isArray(data) ? data : []))
      .catch(() => setFixtures([]));
  }, [grade, round]);

  async function handleSubmit(e: React.FormEvent) {
    e.preventDefault();
    setFormError(null);
    if (!grade)         { setFormError("Select a grade."); return; }

    if (!extendedUntil) { setFormError("Set an extended-until date."); return; }

    setSubmitting(true);
    try {
      const res = await fetch("/api/admin/vote-window-overrides", {
        method:  "POST",
        headers: { "Content-Type": "application/json" },
        body:    JSON.stringify({
          competition,
          grade,
          round,
          fixtureId: fixtureId || null,
          extendedUntil,
        }),
      });
      const data = await res.json();
      if (!res.ok) { setFormError(data.error ?? "Failed to save."); return; }
      setGrade(""); setRound(""); setFixtureId(""); setExtendedUntil("");
      await loadOverrides();
    } catch {
      setFormError("Network error. Please try again.");
    } finally {
      setSubmitting(false);
    }
  }

    await fetch(`/api/admin/vote-window-overrides?id=${id}`, { method: "DELETE" });

  }

        <div className={styles.formTitle}>Add / Update Override</div>

          <div className={styles.fieldRow}>

                disabled={!grade}

                disabled={!round}
              >
                {isSuperadmin && <option value="">All matches in this round</option>}

      </div>

  );
}
```

```ts
import { NextRequest, NextResponse } from "next/server";
import { db } from "@/lib/db";
import { fixtures } from "@/db/schema";
import { and, eq, asc, SQL } from "drizzle-orm";
import { logger } from "@/lib/logger";

/**
 * GET /api/fixtures?grade=...&homeTeam=...&round=...
 *
 * grade is always required.
 * homeTeam is optional — omit to get all fixtures for a grade+round (admin use).
 * round is optional filter.
 */
export async function GET(req: NextRequest) {
  const { searchParams } = new URL(req.url);
  const grade    = searchParams.get("grade")?.trim();
  const homeTeam = searchParams.get("homeTeam")?.trim();
  const round    = searchParams.get("round")?.trim();

  if (!grade) {
    return NextResponse.json(
      { error: "grade query param is required." },
      { status: 400 }
    );
  }

  try {
    logger.info("[fixtures] GET", { category: "api", grade, homeTeam: homeTeam ?? null, round: round ?? null });

    const conditions: SQL[] = [eq(fixtures.gradeName, grade)];
    if (homeTeam) conditions.push(eq(fixtures.homeTeamName, homeTeam));
    if (round)    conditions.push(eq(fixtures.roundName, round));

    const rows = await db
      .select()
      .from(fixtures)
      .where(and(...conditions))
      .orderBy(asc(fixtures.matchDate));

    return NextResponse.json(rows);
  } catch (err) {
    logger.error("[fixtures] GET failed", { category: "api", error: String(err), grade, homeTeam, round });
    return NextResponse.json({ error: "Internal server error." }, { status: 500 });
  }
}
```
